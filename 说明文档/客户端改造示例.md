                ValidatedAt = DateTime.Now,
                ExpiresAt = licenseResult.ExpiresAt,
                LicenseType = licenseResult.LicenseType,
                CustomerInfo = licenseResult.CustomerInfo

            var json = Newtonsoft.Json.JsonConvert.SerializeObject(offlineData);
            var encrypted = Convert.ToBase64String(System.Text.Encoding.UTF8.GetBytes(json));

            System.Diagnostics.Debug.WriteLine($"ä¿å­˜ç¦»çº¿è®¸å¯è¯å¤±è´¥: {ex.Message}");
